; !!! абсолютная
in 0 		
stor a, A 	; ввод A
in 1 ; 
stor a, B 	; ввод B
in 2 ; 
stor a, C 	; ввод C	
in 3 ;
stor a, D 	; ввод D
mvi a, 0 	;
stor a, Y	; Y =ff
load b, A 	; загрузил в регистр rB A
sbm b, B 	; rb = A-B
stor b, Y 	; Y =A-B
mum a, Y 	; (rA,rB)*Y = (A-B)^2
stor b, Y 	; Y = (A-B)^2
load b, C 	; rb = C
mui a, 1	; расширили знак
dvm a, D	; rb = C/D
adm b, Y	; rb = C/D + (A-B)^2
stor b, Y	; Y = rb
load a, Y 	;
out 7		; выводим Y в 7 порт
stop		; конец кода, дальше данные
A: .ds 1 ;
B: .ds 1 ;
C: .ds 1 ;
D: .ds 1 ;
Y: .ds 1 ;


; !!! относительная
.base b ; rB - базовый регистр
start: mvi b, start ; относительный адрес в rB
in 0 		
stor a, A 	; ввод A
in 1 ; 
stor a, B 	; ввод B
in 2 ; 
stor a, C 	; ввод C	
in 3 ;
stor a, D 	; ввод D
mvi a, 0 	;
stor a, Y	; Y = 0
push b
load a, A 	; загрузил в регистр ra A
sbm a, B 	; ra = A-B
pop b
stor a, Y 	; Y =A-B
push b
mov b, a	; ra = rb
mur a, a 	; rA*rB = (A-B)^2
mov a, b
pop b
stor a, Y 	; Y = (A-B)^2
load a, D
push a 		;
load a, C
push b		; stack: /D/base
mov b, a	; rb = C
mui a, 1	; расширили знак
dvm a, 2(sp)	; rb = C/D
mov a, b	; ra = rb
pop b
adm a, Y	; rb = C/D + (A-B)^2
stor a, Y	; Y = ra
out 7		; выводим Y в 7 порт
pop a 		; восстановили стэк
stop		; конец кода, дальше данные
A: .ds 1 ;
B: .ds 1 ;
C: .ds 1 ;
D: .ds 1 ;
Y: .ds 1 ;

; !!! с стеком и регистрами
in 0 		; ввод A
mov b, a 	; rb = A
in 1		; ввод B, ra = B
sbr b, a	; rb = rb - ra = A - B	
mov a, b 	; ra = rb
mur a, a 	; rb = ra * rb = (A - B)^2
push b 		; сохранили в стэк
in 2 		; ввод C, ra = C
mov b, a 	; rb = С
in 3		; ввод D
push a 		; сохранмлм в стэк C
mui a, 1	; расширили знак
dvm a, 1(sp); rb = C/D
pop a 		; ra = C
pop a 		; ra = (A - B)^2
adr a, b 	; ra = (A - B)^2 + C/D
out 7		; выводим Y в 7 порт
stop		; конец кода

; !!! с условием
in 0 
stor a, A 
in 1 
stor a, B 
in 2 
stor a, C 	; Ввели данные
mov b, a 	; rb = C
mum a, A 	; rb = C*A
stor b, Y 	; Y = C*A
load a, B 	; ra = B
cmi a, 0 	; сравниваем с нулём
jz print 	; Если B=0 скачок на метку print
load b, A 	; rb = A
mui a,1 	; расширили знак
dvm a, B 	; rB = A/B
adm b, Y 	; rb = A/B + Y
stor b, Y 
print: load a, Y
out 7 
stop 
A: .ds 1  
B: .ds 1 
C: .ds 1 
Y: .ds 1 


; цикл с предусловием
in 0 
stor a,N 
in 1 
stor a,A 
in 2 
stor a,B 		; Ввели данные
cycl: load b,N 	; rb = N (счетчик)
cmi b,0 		; сравнили с нулем
jz exit			; выход из цикла
load b,A 
mui a,1 
dvm a,N 
adm b,Y 
stor b, Y 
load b,B 
mui a,1 
dvm a,N 
adm b,Y 
stor b, Y 	; Вычисляем Y += a\i+b\i
load b,N 
sbi b,1 	; вычли единицу
stor b,N 	; сохранили счетчик
jmp cycl 	; переход на цикл
exit: load a,Y
out 7 
stop 
A: .ds 1
B: .ds 1 
N: .ds 1 
Y: .ds 1
